cos A  + cos B = 2 cos 1/2 (A + B) cos 1/2(A - B)
sin A + sin B =  2 sin 12 (A + B)  cos 1/2 (A - B)

[tex]\sf \dfrac{cos~ 220+ cos~ 100}{sin~ 140 + sin~ 100}=[/tex]

[tex]\sf=\dfrac{2 cos \frac{1}{2}(220+100) cos \frac{1}{2}(220-100)}{2 sin\frac{1}{2}(140+100) cos\frac{1}{2}(140-100)}[/tex]

[tex]\sf=\dfrac{2 cos (160)cos(60)}{2 sin(120) cos(20)}[/tex]

[tex]\sf=\dfrac{2 cos (180 -20) cos (60)}{2 sin(180- 60) cos(20)}[/tex]

[tex]\sf=\dfrac{2 (- cos20)(\frac{1}{2})}{2 sin60~ cos(20)}[/tex]

[tex]\sf=\dfrac{2 (- cos20)(\frac{1}{2})}{2 (\frac{1}{2}\sqrt 3)~ cos(20)}[/tex]

[tex]\sf=\dfrac{- cos(20)}{\sqrt 3~ cos(20)} = - \frac{1}{\sqrt 3 } = -\frac{1}{3}\sqrt 3[/tex]
