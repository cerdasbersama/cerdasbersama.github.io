[tex] \sqrt{a + b - 2 \sqrt{ab} } [/tex]

dengan

[tex]a + b = 16[/tex]

[tex]a = 9 \\ b = 7[/tex]

( a b ) = 63 --> a > b

a = 9 , b = 7

.

√ (a + b  - 2√ab) =  √(√a - √b)²
